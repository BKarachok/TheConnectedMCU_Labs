342 {
343  unsigned char __exp;
344  __asm__ (
345  "mov %0, %C1 \n\t"
346  "lsl %0 \n\t"
347  "mov %0, %D1 \n\t"
348  "rol %0 "
349  : "=r" (__exp)
350  : "r" (__x) );
351  return __exp != 0xff;
352 }
